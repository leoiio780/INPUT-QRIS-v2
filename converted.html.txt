<p>
    let excelData = [];
</p>
<p>
    function parseData(){
</p>
<p>
        const raw = document.getElementById("rawData").value;
</p>
<p>
        const lines = raw.split("\n");
</p>
<p>
        const tbody = document.getElementById("tbody");
</p>
<p>
        tbody.innerHTML = "";
</p>
<p>
        excelData = [];
</p>
<p>
        let no = 1;
</p>
<p>
        let totalSuccess = 0;
</p>
<p>
        let totalPending = 0;
</p>
<p>
        lines.forEach(line =&gt; {
</p>
<p>
            line = line.trim();
</p>
<p>
            // skip header
</p>
<p>
            if(
</p>
<p>
                line.includes("Member Name Tiket No") ||
</p>
<p>
                line.includes("Date Request") ||
</p>
<p>
                line.startsWith("No.")
</p>
<p>
            ){
</p>
<p>
                return;
</p>
<p>
            }
</p>
<p>
            if(!line) return;
</p>
<p>
            if(!line.match(/^\d+\./)) return;
</p>
<p>
            const cols = line.split("\t");
</p>
<p>
            if(cols.length &lt; 10) return;
</p>
<p>
            const dateRequest = (cols[2] || "").trim();
</p>
<p>
            const member = (cols[4] || "").trim();
</p>
<p>
            const rrn = (cols[6] || "").trim();
</p>
<p>
            let nominal = (cols[7] || "").trim().replace(/\./g,'');
</p>
<p>
            const payment = (cols[10] || "").trim();
</p>
<p>
            const status = line.includes("PENDING") ? "PENDING" : "SUCCESS";
</p>
<p>
            // CROSS DATE CHECK
</p>
<p>
            let keterangan = "NORMAL";
</p>
<p>
            if(dateRequest &amp;&amp; payment){
</p>
<p>
                const reqDate = dateRequest.substring(0,10);
</p>
<p>
                const payDate = payment.substring(0,10);
</p>
<p>
                if(reqDate !== payDate){
</p>
<p>
                    keterangan = `CROSS DATE (${reqDate} → ${payDate})`;
</p>
<p>
                }
</p>
<p>
            }
</p>
<p>
            excelData.push({
</p>
<p>
                No: no,
</p>
<p>
                Member: member,
</p>
<p>
                RRN: rrn,
</p>
<p>
                Nominal: Number(nominal),
</p>
<p>
                Status: status,
</p>
<p>
                Keterangan: keterangan
</p>
<p>
            });
</p>
<p>
            if(status === "SUCCESS") totalSuccess++;
</p>
<p>
            if(status === "PENDING") totalPending++;
</p>
<p>
            tbody.innerHTML += `
</p>
<p>
            &lt;tr&gt;
</p>
<p>
                &lt;td&gt;${no}&lt;/td&gt;
</p>
<p>
                &lt;td&gt;${member}&lt;/td&gt;
</p>
<p>
                &lt;td&gt;${rrn}&lt;/td&gt;
</p>
<p>
                &lt;td&gt;${nominal}&lt;/td&gt;
</p>
<p>
                &lt;td class="${status.toLowerCase()}"&gt;${status}&lt;/td&gt;
</p>
<p>
                &lt;td class="${keterangan.includes('CROSS') ? 'cross' : ''}"&gt;
</p>
<p>
                    ${keterangan}
</p>
<p>
                &lt;/td&gt;
</p>
<p>
            &lt;/tr&gt;
</p>
<p>
            `;
</p>
<p>
            no++;
</p>
<p>
        });
</p>
<p>
        document.getElementById("summary").innerHTML = `
</p>
<p>
            Total Data: &lt;b&gt;${excelData.length}&lt;/b&gt;&lt;br&gt;
</p>
<p>
            SUCCESS: &lt;b&gt;${totalSuccess}&lt;/b&gt; |
</p>
<p>
            PENDING: &lt;b&gt;${totalPending}&lt;/b&gt;
</p>
<p>
        `;
</p>
<p>
    }
</p>
<p>
    function downloadExcel(){
</p>
<p>
        if(excelData.length === 0){
</p>
<p>
            alert("Belum ada data");
</p>
<p>
            return;
</p>
<p>
        }
</p>
<p>
        const ws = XLSX.utils.json_to_sheet(excelData);
</p>
<p>
        const wb = XLSX.utils.book_new();
</p>
<p>
        XLSX.utils.book_append_sheet(wb, ws, "QRIS");
</p>
<p>
        XLSX.writeFile(wb, "hasil_qris.xlsx");
</p>
<p>
    }
